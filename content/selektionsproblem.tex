int select(int a[],int l,int r, int k){
   int p= partition(a,l,r);
   if(l+k<p) 
      return select(a,l,p-1,k);
   if(l+k>p) 
      return select(a,p+1,r,k+l-p-1);
   return a[p];
}
\end{verbatim}
